import tkinter as tk import random import threading import time def show_warm_tip(): # 创建窗口 window = tk.Tk() # 窗口全屏 window.attributes('-fullscreen', True) # 获取屏幕宽高 screen_width = window.winfo_screenwidth() screen_height = window.winfo_screenheight() # 提示文字列表（已添加新内容） tips = [ '多喝水哦~', '保持微笑呀', '每天都要元气满满', '记得吃水果', '保持好心情', '好好爱自己', '我想你了', '梦想成真', '期待下一次见面', '金榜题名', '顺顺利利', '早点休息', '愿所有烦恼都消失', '别熬夜', '今天过得开心嘛', '天冷了，多穿衣服' ] # 文字颜色列表 colors = ['#FFB6C1', '#FF69B4', '#DA70D6', '#EE82EE', '#9370DB', '#7B68EE', '#6495ED', '#48D1CC', '#00FFFF', '#7FFFAA', '#98FB98', '#ADFF2F', '#FFFF00', '#FFD700', '#FFA500', '#FF8C00', '#FF4500', '#A52A2A', '#8B4513'] def create_tip(): for _ in range(5): msg = random.choice(tips) color = random.choice(colors) # 随机位置 x = random.randint(0, screen_width) y = random.randint(0, screen_height) # 随机大小 size = random.randint(12, 24) # 随机透明度（0.5-1.0） alpha = random.uniform(0.5, 1.0) # 创建标签 label = tk.Label(window, text=msg, font=('微软雅黑', size), fg=color, bg='white') label.place(x=x, y=y) # 设置透明度 window.attributes('-alpha', alpha) # 随机显示时间（1-3秒） time.sleep(random.uniform(1, 3)) label.destroy() # 随机间隔时间（0.5-1秒） time.sleep(random.uniform(0.5, 1)) # 多线程创建提示 for _ in range(5): t = threading.Thread(target=create_tip) t.daemon = True t.start() # 线程间间隔时间 time.sleep(0.5) # 窗口循环 window.mainloop() if __name__ == '__main__': show_warm_tip()
